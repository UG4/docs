120 template<typename Matrix_type, typename postype>
121 void WriteMatrix(std::ostream &file, const Matrix_type &A, postype *positions, int dimensions)
122 {
123  PROFILE_FUNC_GROUP("debug");
124  size_t rows = A.num_rows();
125 
126  WriteGridHeader(file, positions, rows, dimensions);
127  PROGRESS_START(prog, rows, "WriteMatrixToConnectionViewer " << dimensions << "d, " << rows << "x" << rows);
128 
129  // write connections
130  for(size_t i=0; i < rows; i++)
131  {
132  PROGRESS_UPDATE(prog, i);
133  for(typename Matrix_type::const_row_iterator conn = A.begin_row(i); conn != A.end_row(i); ++conn)
134  if(conn.value() != 0.0)
135  file << i << " " << conn.index() << " " << conn.value() << std::endl;
136  else
137  file << i << " " << conn.index() << " 0" << std::endl;
138  }
139 }

317 template <typename Matrix_type, typename postype>
318 bool WriteMatrix( std::string filename,
319  const Matrix_type &A,
320  const std::vector<postype> &positionsFrom,
321  const std::vector<postype> &positionsTo, size_t dimensions)
322 {
323  PROFILE_FUNC_GROUP("debug");
324 
325  if(positionsFrom.size() != A.num_cols())
326  {
327  UG_LOG("uFrom.size() != A.num_cols() !\n");
328  return false;
329  }
330  if(positionsTo.size() != A.num_rows())
331  {
332  UG_LOG("uTo.size() != A.num_rows() !\n");
333  return false;
334  }
335 
336  size_t fromOffset = positionsTo.size();
337 
338 
339  std::fstream file(filename.c_str(), std::ios::out);
340  file << CONNECTION_VIEWER_VERSION << std::endl;
341  file << dimensions << std::endl;
342 
343  // write positions
344  file << positionsFrom.size() + positionsTo.size() << std::endl;
345 
346  if(dimensions == 1){
347  for(size_t i=0; i < positionsTo.size(); i++)
348  file << positionsTo[i][0] << " 0.0" << std::endl;
349  for(size_t i=0; i < positionsFrom.size(); i++)
350  file << positionsFrom[i][0] << " 0.0" << std::endl;
351  }
352  else if(dimensions == 2){
353  for(size_t i=0; i < positionsTo.size(); i++)
354  file << positionsTo[i][0] << " " << positionsTo[i][1] << std::endl;
355  for(size_t i=0; i < positionsFrom.size(); i++)
356  file << positionsFrom[i][0] << " " << positionsFrom[i][1] << std::endl;
357  }
358  else{
359  for(size_t i=0; i < positionsTo.size(); i++)
360  file << positionsTo[i][0] << " " << positionsTo[i][1] << " " << positionsTo[i][2] << std::endl;
361  for(size_t i=0; i < positionsFrom.size(); i++)
362  file << positionsFrom[i][0] << " " << positionsFrom[i][1] << " " << positionsFrom[i][2] << std::endl;
363  }
364 
365  file << 1 << std::endl; // show all cons
366  // write connections
367 
368  PROGRESS_START(prog, A.num_rows(), "WriteMatrixToConnectionViewer " << dimensions << "d, " << A.num_rows() << "x" << A.num_rows() );
369  for(size_t i=0; i < A.num_rows(); i++)
370  {
371  PROGRESS_UPDATE(prog, i);
372  for(typename Matrix_type::const_row_iterator conn = A.begin_row(i); conn != A.end_row(i); ++conn)
373  if(conn.value() != 0.0)
374  file << i << " " << conn.index() + fromOffset << " " << conn.value() << std::endl;
375  else
376  file << i << " " << conn.index() + fromOffset << " 0" << std::endl;
377  }
378  return true;
379 }

417  << b[i]-(*compareVec)[i] << std::endl;

427 void WriteVector(std::string filename, const Matrix_type &A, const Vector_type &v,
428  postype *positions, int dimensions, const Vector_type *compareVec=NULL)
429 {
430  PROFILE_FUNC_GROUP("debug");
431  if(dimensions != 2)
432  {
433  UG_LOG(__FILE__ << ":" << __LINE__ << " WriteVectorToConnectionViewer: only dimension=2 supported");
434  return;
435  }
436  filename = GetParallelName(A, filename);
437 
438  size_t rows = A.num_rows();
439  std::fstream file(filename.c_str(), std::ios::out);
440  WriteGridHeader(file, positions, rows, dimensions);
441 
442  PROGRESS_START(prog, rows, "WriteVectorToConnectionViewer " << dimensions << "d, " << A.num_rows() << "x" << A.num_rows() );
443  // write connections
444  for(size_t i=0; i < rows; i++)
445  {
446  PROGRESS_UPDATE(prog, i);
447  for(typename Matrix_type::const_row_iterator conn = A.begin_row(i); conn != A.end_row(i); ++conn)
448  if(conn.value() != 0.0)
449  file << i << " " << conn.index() << " "
450  << std::setprecision(std::numeric_limits<number>::digits10 + 1)
451  << conn.value() << std::endl;
452  else
453  file << i << " " << conn.index() << " 0" << std::endl;
454  }
455 
456 
457  std::string nameValues = filename;
458  nameValues.resize(nameValues.find_last_of("."));
459  nameValues.append(".values");
460  file << "v " << nameValues << "\n";
461 
462  std::fstream fileValues(nameValues.c_str(), std::ios::out);
463  if(compareVec == NULL)
464  {
465  for(size_t i=0; i < rows; i++)
466  fileValues << i << " " << v[i] << "\n";
467  }
468  else
469  {
470  typename Vector_type::value_type t;
471  for(size_t i=0; i < rows; i++)
472  {
473  t = v[i];
474  t -= (*compareVec)[i];
475  fileValues << i << " " << t << "\n";
476  }
477  }
478 
479 }
