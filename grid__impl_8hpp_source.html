108 {

110  && geometry_traits<TGeomObj>::BASE_OBJECT_ID != -1,

543 template <class TGeomObj>
544 Edge* Grid::find_edge_in_associated_edges(TGeomObj* obj, const EdgeVertices& ev)
545 {
546  GRID_PROFILE_FUNC();
547 
548  AssociatedEdgeIterator iterEnd = associated_edges_end(obj);
549  for(AssociatedEdgeIterator iter = associated_edges_begin(obj);
550  iter != iterEnd; ++iter)
551  {
552  Edge* e = *iter;
553  if(CompareVertices(e, &ev))
554  return e;
555  }
556  return NULL;
557 }

560 template <class TGeomObj>
561 Face* Grid::find_face_in_associated_faces(TGeomObj* obj, const FaceVertices& fv)
562 {
563  GRID_PROFILE_FUNC();
564 
565  unsigned long key = hash_key(&fv);
566  AssociatedFaceIterator iterEnd = associated_faces_end(obj);
567  for(AssociatedFaceIterator iter = associated_faces_begin(obj);
568  iter != iterEnd; ++iter)
569  {
570  Face* f = *iter;
571  if(key == hash_key(f))
572  {
573  if(CompareVertices(f, &fv))
574  return f;
575  }
576  }
577 
578  return NULL;
579 }
