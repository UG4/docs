48 template<typename TStorage>
49 std::string NonzeroString(const DenseMatrix<TStorage> &A, const char *brackets, const char *seperator, int base, const char *name)
50 {
51  std::stringstream ss;
52  ss << std::setprecision(9);
53  for(size_t r=0; r<A.num_rows(); ++r)
54  {
55  for(size_t c=0; c<A.num_cols(); ++c)
56  if(A(r, c) != 0.0)
57  ss << name << brackets[0] << r+base << ", " << c+base << brackets[1] << " = " << A(r, c) << seperator;
58  ss << "\n";
59  }
60  return ss.str();
61 }
