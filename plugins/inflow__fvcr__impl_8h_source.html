41 template <typename TDomain, typename TAlgebra>
42 NavierStokesInflowFVCR<TDomain,TAlgebra>::
43 NavierStokesInflowFVCR(SmartPtr< NavierStokesFVCR<TDomain> > spMaster)

55 template <typename TDomain, typename TAlgebra>
56 void NavierStokesInflowFVCR<TDomain,TAlgebra>::
57 add(SmartPtr<CplUserData<MathVector<dim>, dim> > user, const char* subsetsBND)
58 {
59  const std::vector<std::string>& vFctName = m_spMaster->symb_fcts();
60  if(vFctName.empty())
61  UG_THROW("NavierStokesInflow::add: Symbolic names for"
62  " velocity and pressure not set. Please set them first.");
63 
64  std::string velNames;
65  for(int i=0;i<TDomain::dim; ++i)
66  {
67  if(i>0) velNames.append(",");
68  velNames.append(vFctName[i]);
69  }
70  m_spDirichletConstraint->add(user.template cast_dynamic<UserData<MathVector<dim>, dim> >(), velNames.c_str(), subsetsBND);
71 }
