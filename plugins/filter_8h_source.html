314  VType localValue;
315  for (int d=0;d<TGridFunction::dim;d++){
316  u->dof_indices(elem, d, dofInd);
317  assignVal(localValue,d,DoFRef(*u,dofInd[0]));
318  }
319  acUHat[elem]=localValue;
320  }
321  }
