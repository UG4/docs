44 template <typename TDomain, typename TAlgebra>

57 template <typename TDomain, typename TAlgebra>
58 void NavierStokesWall<TDomain,TAlgebra>::
59 add(const char* subsetsBND)
60 {
61  if(m_vFctName.empty())
62  UG_THROW("NavierStokesWall::add: Symbolic names for"
63  " velocity and pressure not set. Please set them first.");
64 
65  for(int i = 0; i < TDomain::dim; ++i)
66  {
67  m_spDirichletConstraint->add(0.0, m_vFctName[i].c_str(), subsetsBND);
68  }
69 }
