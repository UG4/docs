101  template <int refDim>
102  inline void evaluate
103  (
104  MathVector<dim> vValue[],
105  const MathVector<dim> vGlobIP[],
106  number time,
107  int si,
108  GridObject* elem,
109  const MathVector<dim> vCornerCoords[],
110  const MathVector<refDim> vLocIP[],
111  const size_t nip,
112  LocalVector* u,
113  const MathMatrix<refDim, dim>* vJT = NULL
114  ) const
115  {
116  // 1. Get the vectors themselves
117  (*m_spVector) (vValue, vGlobIP, time, si, elem, vCornerCoords, vLocIP, nip, u, vJT);
118 
119  // 2. Prepare the data of the element
120  const ReferenceObjectID roid = elem->reference_object_id ();
121  DimReferenceMapping<refDim, dim>& rMapping = ReferenceMappingProvider::get<refDim, dim> (roid);
122  rMapping.update (vCornerCoords);
123 
124  // 3. At every integration point
125  for (size_t i = 0; i < nip; i++)
126  {
127  // 3a. Get the Jacobian
128  MathMatrix<dim, refDim> J;
129  rMapping.jacobian (J, vLocIP[i]);
130  // 3b. Project the vector to the subspace spanned by the columns of the Jacobian
131  OrthogProjectVec (vValue[i], J);
132  }
133  }

136  template <int refDim>
137  void eval_and_deriv
138  (
139  MathVector<dim> vValue[],
140  const MathVector<dim> vGlobIP[],
141  number time,
142  int si,
143  GridObject* elem,
144  const MathVector<dim> vCornerCoords[],
145  const MathVector<refDim> vLocIP[],
146  const size_t nip,
147  LocalVector* u,
148  bool bDeriv,
149  int s,
150  std::vector<std::vector<MathVector<dim> > > vvvDeriv[],
151  const MathMatrix<refDim, dim>* vJT = NULL
152  ) const
153  {
154  // 1. Get the vectors themselves
155  const MathVector<dim>* vVector = m_spVector->values (s);
156 
157  // 2a. Prepare the data of the element
158  const ReferenceObjectID roid = elem->reference_object_id ();
159  DimReferenceMapping<refDim, dim>& rMapping = ReferenceMappingProvider::get<refDim, dim> (roid);
160  rMapping.update (vCornerCoords);
161 
162  // 2b. Check if we should compute the derivatives
163  if (this->zero_derivative ())
164  bDeriv = false;
165  else
166  this->set_zero (vvvDeriv, nip);
167 
168  // 3. At every integration point
169  for (size_t i = 0; i < nip; i++)
170  {
171  // 3a. Get the Jacobian
172  MathMatrix<dim, refDim> J;
173  rMapping.jacobian (J, vLocIP[i]);
174  // 3b. Project the vector to the subspace spanned by the columns of the Jacobian
175  vValue[i] = vVector[i];
176  OrthogProjectVec (vValue[i], J);
177  // 3c. Project the derivatives in the same way
178  if (! bDeriv) continue;
179  for (size_t fct = 0; fct < m_spDVector->num_fct(); fct++)
180  {
181  const MathVector<dim>* vDVector = m_spDVector->deriv (s, i, fct);
182  const size_t c_fct = this->input_common_fct (0, fct);
183  for (size_t sh = 0; sh < this->num_sh (c_fct); sh++)
184  {
185  vvvDeriv[i][c_fct][sh] = vDVector [sh];
186  OrthogProjectVec (vvvDeriv[i][c_fct][sh], J);
187  }
188  }
189  }
190  }
