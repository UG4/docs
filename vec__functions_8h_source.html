63 template<typename T>
64 inline double VecProd(T &v1, T &v2)
65 {
66  ASSERT_EQUAL(v1.size(), v2.size());
67  double s=0;
68  for(size_t i=0; i<v1.size(); i++)
69  s += v1[i]*v2[i];
70  return s;
71 }
