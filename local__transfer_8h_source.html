204  template <typename TElem>
205  void prolongate(TElem* parent,
206  TransferValueAccessor& vValueChild,
207  TransferValueAccessor& vValueParent)
208  {
209  static const int pdim = TElem::dim;
210 
211  // get reference object ids
212  const ReferenceObjectID parentRoid = parent->reference_object_id();
213 
214  // access the values
215  vValueParent.access_closure(parent);
216 
217  // get vertices if required by some prolongation
218  std::vector<MathVector<dim> > vCornerParent;
219  CollectCornerCoordinates(vCornerParent, *parent, *IElemProlongation<TDomain>::m_spDomain);
220 
221  // get local finite element trial spaces
222  const LocalShapeFunctionSet<pdim>& lsfs =
223  LocalFiniteElementProvider::get<pdim>(parentRoid, m_lfeid);
224 
225  // prolongate to all children with dimension pdim and lower
226  switch(pdim){
227  case 3: prolongate<TElem,Volume>(parent, vValueChild, vValueParent, lsfs, vCornerParent, parentRoid);
228  case 2: prolongate<TElem,Face>(parent, vValueChild, vValueParent, lsfs, vCornerParent, parentRoid);
229  case 1: prolongate<TElem,Edge>(parent, vValueChild, vValueParent, lsfs, vCornerParent, parentRoid);
230  prolongate<TElem,Vertex>(parent, vValueChild, vValueParent, lsfs, vCornerParent, parentRoid);
231  break;
232  default: UG_THROW("Dimension "<<pdim<<" not supported");
233  }
234  }

236  template <typename TParent, typename TChild>
237  void prolongate(TParent* parent,
238  TransferValueAccessor& vValueChild,
239  TransferValueAccessor& vValueParent,
240  const LocalShapeFunctionSet<TParent::dim>& lsfs,
241  std::vector<MathVector<dim> > vCornerParent,
242  const ReferenceObjectID parentRoid)
243  {
244  const MultiGrid* mg = IElemProlongation<TDomain>::m_spGrid.get();
245  static const int pdim = TParent::dim;
246 
247  // number of children of the base type
248  const int numChild = mg->num_children<TChild>(parent);
249 
250  // loop all children
251  for(int c = 0; c < numChild; ++c)
252  {
253  // get child
254  TChild* child = mg->get_child<TChild>(parent, c);
255 
256  // access the values
257  vValueChild.access_inner(child);
258 
259  // global positions of fine dofs
260  std::vector<MathVector<dim> > vDoFPos;
261  InnerDoFPosition(vDoFPos, child, *IElemProlongation<TDomain>::m_spDomain, m_lfeid);
262 
263  // get Reference Mapping
264  DimReferenceMapping<pdim, dim>& map =
265  ReferenceMappingProvider::get<pdim, dim>(parentRoid, vCornerParent);
266 
267  // get local position of DoF
268  std::vector<MathVector<pdim> > vLocPos(vDoFPos.size(), 0.0);
269  map.global_to_local(vLocPos, vDoFPos);
270 
271  // evaluate coarse shape fct at fine local point
272  std::vector<std::vector<number> > vvShape;
273  lsfs.shapes(vvShape, vLocPos);
274 
275  // sum up interpolated values
276  for(size_t csh = 0; csh < vValueChild.size(); ++csh){
277  vValueChild[csh] = 0.0;
278  for(size_t psh = 0; psh < vValueParent.size(); ++psh)
279  vValueChild[csh] += vvShape[csh][psh] * vValueParent[psh];

281  }

315  template <typename TElem>
316  void do_restrict(TElem* parent,
317  TransferValueAccessor& vValueChild,
318  TransferValueAccessor& vValueParent)
319  {
320  // access the values
321  vValueParent.access_inner(parent);
322 
323  // global positions of fine dofs
324  std::vector<MathVector<dim> > vDoFPos;
325  InnerDoFPosition(vDoFPos, parent, *IElemRestriction<TDomain>::m_spDomain, m_lfeid);
326 
327  const MultiGrid* mg = IElemRestriction<TDomain>::m_spGrid.get();
328 
329  // number of children of the base type
330  const int numChild = mg->num_children<TElem>(parent);
331 
332  // loop all children
333  for(size_t ip = 0; ip < vDoFPos.size(); ++ip)
334  {
335  int c = 0;
336  for(; c < numChild; ++c)
337  {
338  // get child
339  TElem* child = mg->get_child<TElem>(parent, c);
340 
341  // global positions of fine dofs
342  std::vector<MathVector<dim> > vChildDoFPos;
343  DoFPosition(vChildDoFPos, child, *IElemRestriction<TDomain>::m_spDomain, m_lfeid);
344 
345  // check if pos matches
346  int cip = -1;
347  for(size_t i = 0; i < vChildDoFPos.size(); ++i)
348  if(VecDistanceSq(vChildDoFPos[i], vDoFPos[ip]) < 1e-8)
349  cip = i;
350 
351  // if not found in this child, continue
352  if(cip == -1) continue;
353 
354  // access the values
355  vValueChild.access_closure(child);
356 
357  // sum up interpolated values
358  vValueParent[ip] = vValueChild[cip];
359 
360  break;
361  }
362  // check that DoF found
363  if(c == numChild)
364  UG_THROW("Lagrange-Element-Coarsening: Cannot find DoF position "
365  "in child elems for "<<ip<<"'s Pos: "<<vDoFPos[ip])
366  }
367 
368  }

425  TransferValueAccessor& vValueParent)
426  {
427  const MultiGrid* mg = IElemProlongation<TDomain>::m_spGrid.get();
428  // a) prolongation from a side
429  if(TParent::dim == m_lfeid.dim()-1){
430  typedef typename TParent::sideof TElem;
431  typedef TParent TSide;
432 
433  // get child sides
434  const int numChild = mg->num_children<TParent>(parent);
435  if(numChild == 0) return;

437  // get childs

439  for(int c = 0; c < numChild; ++c)
440  vChildSide[c] = mg->get_child<TParent>(parent, c);
441

443  typename Grid::traits<TElem>::secure_container vElem;
444  const_cast<MultiGrid*>(mg)->associated_elements(vElem, parent);
445  std::vector<TElem*> vParentElem(vElem.size());

456  typedef typename TParent::side TSide;
457 
458  // get child sides
459  const int numChild = mg->num_children<TSide>(parent);
460  if(numChild == 0) return;

464  for(int c = 0; c < numChild; ++c)
