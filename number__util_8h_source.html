120 template<typename T>
121 inline bool WarnIfIsVectorNanOrTooBig(const T &t, const char *callerName)
122 {
123  if(g_bNoNANCheck) return true;
124  size_t iFirst, iCount=0;
125  for(size_t i=0; i<t.size(); i++)
126  if(IsFiniteAndNotTooBig(t[i]) == false)
127  {
128  if(iCount == 0) iFirst = i;
129  iCount++;
130  }
131 
132  if(iCount > 0)
133  {
134  UG_LOG("WARNING in " << callerName << ": " << iFirst << " nan or too big (" << t[iFirst] << "). ");
135  if(iCount > 1)
136  { UG_LOG(iCount-1 << " more.");}
137  UG_LOG("\n");
138  return false;
139  }
