95 template <class TPositionAttachment>
96 bool GridWriterVTU::
97 new_piece(Grid& grid, ISubsetHandler* psh, TPositionAttachment& aPos)
98 {
99  using namespace std;
100 
101  if(m_pieceMode == OPEN)
102  end_piece();
103 
104  m_pieceMode = OPEN;
105  m_pointDataMode = NONE;
106  m_cellDataMode = NONE;
107 
108  ostream& out = out_stream();
109 
110  m_curGrid = &grid;
111  m_curSH = psh;
112 
113  m_pieceSubsetHandlers.clear();
114  m_cells.clear();
115 
116 // if a subset-handler is specified, all grid elements which are assigned to
117 // subsets are considered as cells.
118 // If not, only elements of highest dimension are considered as cells.
119 
120  if(psh){
121  add_subset_handler(*psh, string("regions"));
122  collect_cells<Vertex>(m_cells, grid, IsNotInSubset(*psh, -1));
123  collect_cells<Edge>(m_cells, grid, IsNotInSubset(*psh, -1));
124  collect_cells<Face>(m_cells, grid, IsNotInSubset(*psh, -1));
125  collect_cells<Volume>(m_cells, grid, IsNotInSubset(*psh, -1));
126  }
127  else{
128  if(grid.num<Volume>() > 0)
129  collect_cells<Volume>(m_cells, grid, ConsiderAll());
130  else if(grid.num<Face>() > 0)
131  collect_cells<Face>(m_cells, grid, ConsiderAll());
132  else if(grid.num<Edge>() > 0)
133  collect_cells<Edge>(m_cells, grid, ConsiderAll());
134  else
135  collect_cells<Vertex>(m_cells, grid, ConsiderAll());
136  }
137 
138  out << " <Piece NumberOfPoints=\"" << grid.num<Vertex>() << "\""
139  << " NumberOfCells=\"" << m_cells.size() << "\">" << endl;
140 
141 
142 // write points
143  out << " <Points>" << endl;
144  write_vector_data<Vertex>(grid, aPos, "");
145  out << " </Points>" << endl;
146 
147 // write cells
148 // first we'll assign indices to the vertices, which can then be used as
149 // references by the cells.
150  AInt aInd;
151  grid.attach_to_vertices(aInd);
152  Grid::VertexAttachmentAccessor<AInt> aaInd(grid, aInd);
153  AssignIndices(grid.begin<Vertex>(), grid.end<Vertex>(), aaInd, 0);
154 
155  write_cells(m_cells, grid, aaInd);
156 
157  grid.detach_from_vertices(aInd);
158 
159  return true;
160 }

210 template <class TPositionAttachment>
211 bool GridReaderVTU::
212 grid(Grid& gridOut, size_t index, TPositionAttachment& aPos)
213 {
214  using namespace rapidxml;
215  using namespace std;
216 
217 // make sure that a node at the given index exists
218  if(num_grids() <= index){
219  UG_LOG(" GridReaderVTU::read: bad grid index!\n");
220  return false;
221  }
222 
223  Grid& grid = gridOut;
224 
225 // Since we have to create all elements in the correct order and
226 // since we have to make sure that no elements are created in between,
227 // we'll first disable all grid-options and reenable them later on
228  uint gridopts = grid.get_options();
229  grid.set_options(GRIDOPT_NONE);
230 
231 // access node data
232  if(!grid.has_vertex_attachment(aPos)){
233  grid.attach_to_vertices(aPos);
234  }
235 
236  Grid::VertexAttachmentAccessor<TPositionAttachment> aaPos(grid, aPos);
237 
238 // store the grid in the grid-vector and assign indices to the vertices
239  m_entries[index].grid = &grid;
240 
241 // get the grid-node and the vertex-vector
242  xml_node<>* gridNode = m_entries[index].node;
243  vector<Vertex*>& vertices = m_entries[index].vertices;
244  vector<GridObject*>& cells = m_entries[index].cells;
245 
246 // iterate over all pieces
247  xml_node<>* pieceNode = gridNode;
248 // first we'll create all points and cells, then we'll parse point- and cell-data
249  xml_node<>* pointsNode = pieceNode->first_node("Points");
250  UG_COND_THROW(pointsNode == NULL, "Missing Points node in UnstructuredGrid node!")
251 
252  size_t vrtOffset = vertices.size();
253  create_vertices(vertices, grid, pointsNode, aaPos);
254 
255  xml_node<>* cellsNode = pieceNode->first_node("Cells");
256  UG_COND_THROW(cellsNode == NULL, "Missing Cells node in UnstructuredGrid node!")
257  create_cells(cells, grid, cellsNode, vertices, vrtOffset);
258 
259 // reenable the grids options.
260  grid.set_options(gridopts);
261 
262  return true;
263 }

265 template <class TAAPos>
266 bool GridReaderVTU::
267 create_vertices(std::vector<Vertex*>& vrtsOut, Grid& grid,
268  rapidxml::xml_node<>* vrtNode, TAAPos aaPos)
269 {
270  using namespace rapidxml;
271  using namespace std;
272 
273  xml_node<>* dataNode = vrtNode->first_node("DataArray");
274  UG_COND_THROW(!dataNode, "Missing DataArray node in Points node");
275 
276  int numSrcCoords = -1;
277  xml_attribute<>* attrib = dataNode->first_attribute("NumberOfComponents");
278  if(attrib)
279  numSrcCoords = atoi(attrib->value());
280 
281  int numDestCoords = (int)TAAPos::ValueType::Size;
282 
283  assert(numDestCoords > 0 && "bad position attachment type");
284 
285  if(numSrcCoords < 1 || numDestCoords < 1)
286  return false;
287 
288 // create a buffer with which we can access the data
289  string str(dataNode->value(), dataNode->value_size());
290  stringstream ss(str, ios_base::in);
291 
292 // if numDestCoords == numSrcCoords parsing will be faster
293  if(numSrcCoords == numDestCoords){
294  while(!ss.eof()){
295  // read the data
296  typename TAAPos::ValueType v;
297 
298  for(int i = 0; i < numSrcCoords; ++i)
299  ss >> v[i];
300 
301  // make sure that everything went right
302  if(ss.fail())
303  break;
304 
305  // create a new vertex
306  RegularVertex* vrt = *grid.create<RegularVertex>();
307  vrtsOut.push_back(vrt);
308 
309  // set the coordinates
310  aaPos[vrt] = v;
311  }
312  }
313  else{
314  // we have to be careful with reading.
315  // if numDestCoords < numSrcCoords we'll ignore some coords,
316  // in the other case we'll add some 0's.
317  int minNumCoords = min(numSrcCoords, numDestCoords);
318  typename TAAPos::ValueType::value_type dummy = 0;
319 
320  while(!ss.eof()){
321  // read the data
322  typename TAAPos::ValueType v;
323 
324  int iMin;
325  for(iMin = 0; iMin < minNumCoords; ++iMin)
326  ss >> v[iMin];
327 
328  // ignore unused entries in the input buffer
329  for(int i = iMin; i < numSrcCoords; ++i)
330  ss >> dummy;
331 
332  // add 0's to the vector
333  for(int i = iMin; i < numDestCoords; ++i)
334  v[i] = 0;
335 
336  // make sure that everything went right
337  if(ss.fail()){
338  UG_LOG("GridReaderVTU::create_vertices: Failed to read vertex.\n");
339  return false;
340  }
341 
342  // create a new vertex
343  RegularVertex* vrt = *grid.create<RegularVertex>();
344  vrtsOut.push_back(vrt);
345 
346  // set the coordinates
347  aaPos[vrt] = v;
348  }
349  }
350 
351  return true;
352 }
