57 template <class TAPosition>
58 bool LoadGridFromUGX(Grid& grid, ISubsetHandler& sh, const char* filename,
59  TAPosition& aPos)
60 {
61  GridReaderUGX ugxReader;
62  if(!ugxReader.parse_file(filename)){
63  UG_LOG("ERROR in LoadGridFromUGX: File not found: " << filename << std::endl);
64  return false;
65  }
66 
67  if(ugxReader.num_grids() < 1){
68  UG_LOG("ERROR in LoadGridFromUGX: File contains no grid.\n");
69  return false;
70  }
71 
72  ugxReader.grid(grid, 0, aPos);
73 
74  if(ugxReader.num_subset_handlers(0) > 0)
75  ugxReader.subset_handler(sh, 0, 0);
76 
77  return true;
78 }

82 template <class TPositionAttachment>
83 bool GridWriterUGX::
84 add_grid(Grid& grid, const char* name,
85  TPositionAttachment& aPos)
86 {
87  using namespace rapidxml;
88  using namespace std;
89 // access node data
90  if(!grid.has_vertex_attachment(aPos)){
91  UG_LOG(" position attachment missing in grid " << name << endl);
92  return false;
93  }
94 
95  Grid::VertexAttachmentAccessor<TPositionAttachment> aaPos(grid, aPos);
96 
97 // create a new grid-node
98  xml_node<>* gridNode = m_doc.allocate_node(node_element, "grid");
99  gridNode->append_attribute(m_doc.allocate_attribute("name", name));
100 
101 // store the grid and the node in an entry
102  m_vEntries.push_back(Entry(&grid, gridNode));
103 
104 // append it to the document
105  m_doc.append_node(gridNode);
106 
107 // initialize the grids attachments
108  init_grid_attachments(grid);
109 
110 // access indices
111  Grid::EdgeAttachmentAccessor<AInt> aaIndEDGE(grid, m_aInt);
112  Grid::FaceAttachmentAccessor<AInt> aaIndFACE(grid, m_aInt);
113 
114 // write vertices
115  if(grid.num<RegularVertex>() > 0)
116  gridNode->append_node(create_vertex_node(grid.begin<RegularVertex>(),
117  grid.end<RegularVertex>(), aaPos));
118 
119 // write constrained vertices
120  if(grid.num<ConstrainedVertex>() > 0)
121  gridNode->append_node(create_constrained_vertex_node(
122  grid.begin<ConstrainedVertex>(),
123  grid.end<ConstrainedVertex>(),
124  aaPos, aaIndEDGE, aaIndFACE));
125 
126 // add the remaining grid elements to the nodes
127  add_elements_to_node(gridNode, grid);
128 
129  process_global_attachments<Vertex>(grid, gridNode);
130  process_global_attachments<Edge>(grid, gridNode);
131  process_global_attachments<Face>(grid, gridNode);
132  process_global_attachments<Volume>(grid, gridNode);
133 
134  return true;
135 }

156 template <class TElem, class TAttachment>
157 void GridWriterUGX::
158 add_attachment(TAttachment attachment,
159  const char* name,
160  size_t refGridIndex)
161 {
162  using namespace rapidxml;
163  using namespace std;
164 
165  UG_COND_THROW(refGridIndex >= m_vEntries.size(),
166  "Invalid refGridIndex: " << refGridIndex
167  << ", but only " << m_vEntries.size() << " grids available.");
168 
169  Grid& grid = *m_vEntries[refGridIndex].grid;
170  stringstream ss;
171 
172  if(!grid.has_attachment<TElem>(attachment))
173  return;
174 
175  Grid::AttachmentAccessor<TElem, TAttachment> aaVal(grid, attachment);
176 
177  const typename Grid::traits<TElem>::iterator iterEnd = grid.end<TElem>();
178  for(typename Grid::traits<TElem>::iterator iter = grid.begin<TElem>();
179  iter != iterEnd;)
180  {
181  attachment_io_traits<TAttachment>::write_value(ss, aaVal[*iter]);
182  UG_COND_THROW(!ss, "Failed to write attachment entry.\n");
183  ++iter;
184  if(iter != iterEnd)
185  ss << " ";
186  }
187 
188 // create the node
189  xml_node<>* node = m_doc.allocate_node(
190  node_element,
191  attachment_node_name<TElem>(),
192  m_doc.allocate_string(ss.str().c_str()));
193 
194 // attributes
195  node->append_attribute(
196  m_doc.allocate_attribute(
197  "name",
198  m_doc.allocate_string(name)));
199 
200  node->append_attribute(
201  m_doc.allocate_attribute(
202  "type",
203  m_doc.allocate_string(
204  attachment_info_traits<TAttachment>::type_name().c_str())));
205 
206 //todo: only default pass-on is stored, not the actual used one!
207  node->append_attribute(
208  m_doc.allocate_attribute(
209  "passOn",
210  m_doc.allocate_string(mkstr(int(attachment.default_pass_on_behaviour())).c_str())));
211 
212  m_vEntries[refGridIndex].node->append_node(node);
213 }

216 template <class TAAPos>
217 rapidxml::xml_node<>*
218 GridWriterUGX::
219 create_vertex_node(RegularVertexIterator vrtsBegin,
220  RegularVertexIterator vrtsEnd,
221  TAAPos& aaPos)
222 {
223  using namespace rapidxml;
224  using namespace std;
225 // the number of coordinates
226  const int numCoords = (int)TAAPos::ValueType::Size;
227 
228 // write the vertices to a temporary stream
229  stringstream ss;
230  ss.precision(18);
231  for(RegularVertexIterator iter = vrtsBegin; iter != vrtsEnd; ++iter)
232  {
233  for(int i = 0; i < numCoords; ++i)
234  ss << aaPos[*iter][i] << " ";
235  }
236 
237 // create the node
238  xml_node<>* node = NULL;
239 
240  if(ss.str().size() > 0){
241  // allocate a string and erase last character(' ')
242  char* nodeData = m_doc.allocate_string(ss.str().c_str(), ss.str().size());
243  nodeData[ss.str().size()-1] = 0;
244  // create a node with some data
245  node = m_doc.allocate_node(node_element, "vertices", nodeData);
246  }
247  else{
248  // create an emtpy node
249  node = m_doc.allocate_node(node_element, "vertices");
250  }
251 
252  char* buff = m_doc.allocate_string(NULL, 10);
253  sprintf(buff, "%d", numCoords);
254  node->append_attribute(m_doc.allocate_attribute("coords", buff));
255 
256 // return the node
257  return node;
258 }

260 template <class TAAPos>
261 rapidxml::xml_node<>*
262 GridWriterUGX::
263 create_constrained_vertex_node(ConstrainedVertexIterator vrtsBegin,
264  ConstrainedVertexIterator vrtsEnd,
265  TAAPos& aaPos,
266  AAEdgeIndex aaIndEDGE,
267  AAFaceIndex aaIndFACE)
268 {
269  using namespace rapidxml;
270  using namespace std;
271 // the number of coordinates
272  const int numCoords = (int)TAAPos::ValueType::Size;
273 
274 // write the vertices to a temporary stream
275  stringstream ss;
276  ss.precision(18);
277  for(ConstrainedVertexIterator iter = vrtsBegin; iter != vrtsEnd; ++iter)
278  {
279  for(int i = 0; i < numCoords; ++i)
280  ss << aaPos[*iter][i] << " ";
281 
282  // write index and local coordinate of associated constraining element
283  // codes: -1: no constraining element
284  // 0: vertex. index follows (not yet supported)
285  // 1: edge. index and 1 local coordinate follow.
286  // 2: face. index and 2 local coordinates follow.
287  // 3: volume. index and 3 local coordinates follow. (not yet supported)
288  Edge* ce = dynamic_cast<Edge*>((*iter)->get_constraining_object());
289  Face* cf = dynamic_cast<Face*>((*iter)->get_constraining_object());
290  if(ce)
291  ss << "1 " << aaIndEDGE[ce] << " " << (*iter)->get_local_coordinate_1() << " ";
292  else if(cf)
293  ss << "2 " << aaIndFACE[cf] << " " << (*iter)->get_local_coordinate_1()
294  << " " << (*iter)->get_local_coordinate_2() << " ";
295  else
296  ss << "-1 ";
297  }
298 
299 // create the node
300  xml_node<>* node = NULL;
301 
302  if(ss.str().size() > 0){
303  // allocate a string and erase last character(' ')
304  char* nodeData = m_doc.allocate_string(ss.str().c_str(), ss.str().size());
305  nodeData[ss.str().size()-1] = 0;
306  // create a node with some data
307  node = m_doc.allocate_node(node_element, "constrained_vertices", nodeData);
308  }
309  else{
310  // create an emtpy node
311  node = m_doc.allocate_node(node_element, "constrained_vertices");
312  }
313 
314  char* buff = m_doc.allocate_string(NULL, 10);
315  sprintf(buff, "%d", numCoords);
316  node->append_attribute(m_doc.allocate_attribute("coords", buff));
317 
318 // return the node
319  return node;
320 }

323 template <class TElem>
324 void GridWriterUGX::
325 process_global_attachments(Grid& grid, rapidxml::xml_node<>* gridNode)
326 {
327  using namespace std;
328  using namespace rapidxml;
329  const vector<string>& attachmentNames = GlobalAttachments::declared_attachment_names();
330 
331  for(size_t ia = 0; ia < attachmentNames.size(); ++ia){
332  const std::string& name = attachmentNames[ia];
333  if(!GlobalAttachments::is_attached<TElem>(grid, name))
334  continue;
335 
336  stringstream ss;
337  GlobalAttachments::write_attachment_values<TElem>(ss, grid, name);
338 
339  // create the node
340  xml_node<>* node = m_doc.allocate_node(
341  node_element,
342  attachment_node_name<TElem>(),
343  m_doc.allocate_string(ss.str().c_str()));
344 
345  // attributes
346  node->append_attribute(
347  m_doc.allocate_attribute(
348  "name",
349  m_doc.allocate_string(name.c_str())));
350 
351  node->append_attribute(
352  m_doc.allocate_attribute(
353  "type",
354  m_doc.allocate_string(
355  GlobalAttachments::type_name(name))));
356 
357  node->append_attribute(
358  m_doc.allocate_attribute(
359  "passOn",
360  m_doc.allocate_string(mkstr(int(
361  GlobalAttachments::attachment_pass_on_behaviour(name))).c_str())));
362 
363  node->append_attribute(
364  m_doc.allocate_attribute(
365  "global",
366  m_doc.allocate_string("1")));
367 
368  gridNode->append_node(node);
369  }
370 }

375 // implementation of GridReaderUGX
376 template <class TPositionAttachment>
377 bool GridReaderUGX::
378 grid(Grid& gridOut, size_t index,
379  TPositionAttachment& aPos)
380 {
381  using namespace rapidxml;
382  using namespace std;
383 
384 // make sure that a node at the given index exists
385  if(num_grids() <= index){
386  UG_LOG(" GridReaderUGX::read: bad grid index!\n");
387  return false;
388  }
389 
390  Grid& grid = gridOut;
391 
392 // Since we have to create all elements in the correct order and
393 // since we have to make sure that no elements are created in between,
394 // we'll first disable all grid-options and reenable them later on
395  uint gridopts = grid.get_options();
396  grid.set_options(GRIDOPT_NONE);
397 
398 // access node data
399  if(!grid.has_vertex_attachment(aPos)){
400  grid.attach_to_vertices(aPos);
401  }
402 
403  Grid::VertexAttachmentAccessor<TPositionAttachment> aaPos(grid, aPos);
404 
405 // store the grid in the grid-vector and assign indices to the vertices
406  m_entries[index].grid = &grid;
407 
408 // get the grid-node and the vertex-vector
409  xml_node<>* gridNode = m_entries[index].node;
410  vector<Vertex*>& vertices = m_entries[index].vertices;
411  vector<Edge*>& edges = m_entries[index].edges;
412  vector<Face*>& faces = m_entries[index].faces;
413  vector<Volume*>& volumes = m_entries[index].volumes;
414 
415 // we'll record constraining objects for constrained-vertices and constrained-edges
416  std::vector<std::pair<int, int> > constrainingObjsVRT;
417  std::vector<std::pair<int, int> > constrainingObjsEDGE;
418  std::vector<std::pair<int, int> > constrainingObjsTRI;
419  std::vector<std::pair<int, int> > constrainingObjsQUAD;
420 
421 // iterate through the nodes in the grid and create the entries
422  xml_node<>* curNode = gridNode->first_node();
423  for(;curNode; curNode = curNode->next_sibling()){
424  bool bSuccess = true;
425  const char* name = curNode->name();
426  if(strcmp(name, "vertices") == 0)
427  bSuccess = create_vertices(vertices, grid, curNode, aaPos);
428  else if(strcmp(name, "constrained_vertices") == 0)
429  bSuccess = create_constrained_vertices(vertices, constrainingObjsVRT,
430  grid, curNode, aaPos);
431  else if(strcmp(name, "edges") == 0)
432  bSuccess = create_edges(edges, grid, curNode, vertices);
433  else if(strcmp(name, "constraining_edges") == 0)
434  bSuccess = create_constraining_edges(edges, grid, curNode, vertices);
435  else if(strcmp(name, "constrained_edges") == 0)
436  bSuccess = create_constrained_edges(edges, constrainingObjsEDGE,
437  grid, curNode, vertices);
438  else if(strcmp(name, "triangles") == 0)
439  bSuccess = create_triangles(faces, grid, curNode, vertices);
440  else if(strcmp(name, "constraining_triangles") == 0)
441  bSuccess = create_constraining_triangles(faces, grid, curNode, vertices);
442  else if(strcmp(name, "constrained_triangles") == 0)
443  bSuccess = create_constrained_triangles(faces, constrainingObjsTRI,
444  grid, curNode, vertices);
445  else if(strcmp(name, "quadrilaterals") == 0)
446  bSuccess = create_quadrilaterals(faces, grid, curNode, vertices);
447  else if(strcmp(name, "constraining_quadrilaterals") == 0)
448  bSuccess = create_constraining_quadrilaterals(faces, grid, curNode, vertices);
449  else if(strcmp(name, "constrained_quadrilaterals") == 0)
450  bSuccess = create_constrained_quadrilaterals(faces, constrainingObjsQUAD,
451  grid, curNode, vertices);
452  else if(strcmp(name, "tetrahedrons") == 0)
453  bSuccess = create_tetrahedrons(volumes, grid, curNode, vertices);
454  else if(strcmp(name, "hexahedrons") == 0)
455  bSuccess = create_hexahedrons(volumes, grid, curNode, vertices);
456  else if(strcmp(name, "prisms") == 0)
457  bSuccess = create_prisms(volumes, grid, curNode, vertices);
458  else if(strcmp(name, "pyramids") == 0)
459  bSuccess = create_pyramids(volumes, grid, curNode, vertices);
460  else if(strcmp(name, "octahedrons") == 0)
461  bSuccess = create_octahedrons(volumes, grid, curNode, vertices);
462 
463  else if(strcmp(name, "vertex_attachment") == 0)
464  bSuccess = read_attachment<Vertex>(grid, curNode);
465  else if(strcmp(name, "edge_attachment") == 0)
466  bSuccess = read_attachment<Edge>(grid, curNode);
467  else if(strcmp(name, "face_attachment") == 0)
468  bSuccess = read_attachment<Face>(grid, curNode);
469  else if(strcmp(name, "volume_attachment") == 0)
470  bSuccess = read_attachment<Volume>(grid, curNode);
471 
472 
473  if(!bSuccess){
474  grid.set_options(gridopts);
475  return false;
476  }
477  }
478 
479 // resolve constrained object relations
480  if(!constrainingObjsVRT.empty()){
481  //UG_LOG("num-edges: " << edges.size() << std::endl);
482  // iterate over the pairs.
483  // at the same time we'll iterate over the constrained vertices since
484  // they are synchronized.
485  ConstrainedVertexIterator hvIter = grid.begin<ConstrainedVertex>();
486  for(std::vector<std::pair<int, int> >::iterator iter = constrainingObjsVRT.begin();
487  iter != constrainingObjsVRT.end(); ++iter, ++hvIter)
488  {
489  ConstrainedVertex* hv = *hvIter;
490 
491  switch(iter->first){
492  case 1: // constraining object is an edge
493  {
494  // make sure that the index is valid
495  if(iter->second >= 0 && iter->second < (int)edges.size()){
496  // get the edge
497  ConstrainingEdge* edge = dynamic_cast<ConstrainingEdge*>(edges[iter->second]);
498  if(edge){
499  hv->set_constraining_object(edge);
500  edge->add_constrained_object(hv);
501  }
502  else{
503  UG_LOG("WARNING: Type-ID / type mismatch. Ignoring edge " << iter->second << ".\n");
504  }
505  }
506  else{
507  UG_LOG("ERROR in GridReaderUGX: Bad edge index in constrained vertex: " << iter->second << "\n");
508  }
509  }break;
510 
511  case 2: // constraining object is an face
512  {
513  // make sure that the index is valid
514  if(iter->second >= 0 && iter->second < (int)faces.size()){
515  // get the edge
516  ConstrainingFace* face = dynamic_cast<ConstrainingFace*>(faces[iter->second]);
517  if(face){
518  hv->set_constraining_object(face);
519  face->add_constrained_object(hv);
520  }
521  else{
522  UG_LOG("WARNING in GridReaderUGX: Type-ID / type mismatch. Ignoring face " << iter->second << ".\n");
523  }
524  }
525  else{
526  UG_LOG("ERROR in GridReaderUGX: Bad face index in constrained vertex: " << iter->second << "\n");
527  }
528  }break;
529 
530  default:
531  {
532 // UG_LOG("WARNING in GridReaderUGX: unsupported type-id of constraining vertex"
533 // << " at " << GetGridObjectCenter(grid, hv) << "\n");
534  break;
535  }
536  }
537  }
538  }
539 
540  if(!constrainingObjsEDGE.empty()){
541  // iterate over the pairs.
542  // at the same time we'll iterate over the constrained vertices since
543  // they are synchronized.
544  ConstrainedEdgeIterator ceIter = grid.begin<ConstrainedEdge>();
545  for(std::vector<std::pair<int, int> >::iterator iter = constrainingObjsEDGE.begin();
546  iter != constrainingObjsEDGE.end(); ++iter, ++ceIter)
547  {
548  ConstrainedEdge* ce = *ceIter;
549 
550  switch(iter->first){
551  case 1: // constraining object is an edge
552  {
553  // make sure that the index is valid
554  if(iter->second >= 0 && iter->second < (int)edges.size()){
555  // get the edge
556  ConstrainingEdge* edge = dynamic_cast<ConstrainingEdge*>(edges[iter->second]);
557  if(edge){
558  ce->set_constraining_object(edge);
559  edge->add_constrained_object(ce);
560  }
561  else{
562  UG_LOG("WARNING in GridReaderUGX: Type-ID / type mismatch. Ignoring edge " << iter->second << ".\n");
563  }
564  }
565  else{
566  UG_LOG("ERROR in GridReaderUGX: Bad edge index in constrained edge.\n");
567  }
568  }break;
569  case 2: // constraining object is an face
570  {
571  // make sure that the index is valid
572  if(iter->second >= 0 && iter->second < (int)faces.size()){
573  // get the edge
574  ConstrainingFace* face = dynamic_cast<ConstrainingFace*>(faces[iter->second]);
575  if(face){
576  ce->set_constraining_object(face);
577  face->add_constrained_object(ce);
578  }
579  else{
580  UG_LOG("WARNING in GridReaderUGX: Type-ID / type mismatch. Ignoring face " << iter->second << ".\n");
581  }
582  }
583  else{
584  UG_LOG("ERROR in GridReaderUGX: Bad face index in constrained edge: " << iter->second << "\n");
585  }
586  }break;
587 
588  default:
589  {
590 // UG_LOG("WARNING in GridReaderUGX: unsupported type-id of constraining edge"
591 // << " at " << GetGridObjectCenter(grid, ce) << "\n");
592  break;
593  }
594  }
595  }
596  }
597 
598  if(!constrainingObjsTRI.empty()){
599  // iterate over the pairs.
600  // at the same time we'll iterate over the constrained vertices since
601  // they are synchronized.
602  ConstrainedTriangleIterator cfIter = grid.begin<ConstrainedTriangle>();
603  for(std::vector<std::pair<int, int> >::iterator iter = constrainingObjsTRI.begin();
604  iter != constrainingObjsTRI.end(); ++iter, ++cfIter)
605  {
606  ConstrainedFace* cdf = *cfIter;
607 
608  switch(iter->first){
609  case 2: // constraining object is an face
610  {
611  // make sure that the index is valid
612  if(iter->second >= 0 && iter->second < (int)faces.size()){
613  // get the edge
614  ConstrainingFace* face = dynamic_cast<ConstrainingFace*>(faces[iter->second]);
615  if(face){
616  cdf->set_constraining_object(face);
617  face->add_constrained_object(cdf);
618  }
619  else{
620  UG_LOG("WARNING in GridReaderUGX: Type-ID / type mismatch. Ignoring face " << iter->second << ".\n");
621  }
622  }
623  else{
624  UG_LOG("ERROR in GridReaderUGX: Bad face index in constrained face: " << iter->second << "\n");
625  }
626  }break;
627 
628  default:
629  {
630 // UG_LOG("WARNING in GridReaderUGX: unsupported type-id of constraining triangle"
631 // << " at " << GetGridObjectCenter(grid, cdf) << "\n");
632  break;
633  }
634  }
635  }
636  }
637 
638  if(!constrainingObjsQUAD.empty()){
639  // iterate over the pairs.
640  // at the same time we'll iterate over the constrained vertices since
641  // they are synchronized.
642  ConstrainedQuadrilateralIterator cfIter = grid.begin<ConstrainedQuadrilateral>();
