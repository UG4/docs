function FreeUserData()
for n,v in pairs(_G) do
if type(v) == "userdata" then
_G[n] = nil
end
if type(v) == "table" then
if(n ~= "_G" and n ~= "io" and n ~= "package" and n ~= "gnuplot") then
FreeUserDataInTable(_G[n])
end
end
end
collectgarbage("collect")
end
