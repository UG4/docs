305  nodeType *opr1(int oper, nodeType *op)
306  {
307  nodeType *p;
308  p = new nodeType;
309 
310  p->type = typeOpr;
311  p->opr.oper = oper;
312  p->opr.nops = 1;
313  p->opr.op = new nodeType*[1];
314  p->opr.op[0] = op;
315 
316  if(oper == 'R')
317  {
318  nodeType *aa = op;
319  int i=1;
320  while(aa->type == typeOpr && aa->opr.oper == ',')
321  {
322  i++;
323  aa = aa->opr.op[1];
324  }
325  setRet(i);
326  }
327  return p;
328  }
