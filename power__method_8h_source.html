126  void set_linear_operator_A(SmartPtr<ILinearOperator<vector_type> > loA)
127  {

165  int calculate_max_eigenvalue()
166  {
167  PROFILE_FUNC_GROUP("PowerMethod");
168 
169  UG_COND_THROW(m_spSolver == SPNULL && m_spLinOpB != SPNULL, "PowerMethod::calculate_max_eigenvalue(): Solver not set, please specify.");
170  if(m_spLinOpB != SPNULL)
171  m_spSolver->init(m_spLinOpB);
172 
173  m_spResidual = create_approximation_vector();
174 
175  for(m_iteration = 0; m_iteration < m_maxIterations; ++m_iteration)
176  {
177  m_spEigenvectorOld = m_spEigenvector->clone();
178 
179  // residual = A v
180  UG_COND_THROW(m_spLinOpA == SPNULL, "PowerMethod::calculate_max_eigenvalue(): Linear operator A not set, please specify.");
181  m_spLinOpA->apply(*m_spResidual, *m_spEigenvector);
182 
183  // v = B^-1 A v
184  if(m_spLinOpB != SPNULL)
185  m_spSolver->apply(*m_spEigenvector, *m_spResidual);
186  // in case B is not set: v = A v
187  else
188  m_spEigenvector = m_spResidual->clone();
189 
190  // v = v / ||v||_B
191  normalize_approximations();
192 
193  // in case B is not set, restore storage type of m_spEigenvector to PST_CONSISTENT
194  // changed by norm calculation in normalize_approximations()
195  if(m_spLinOpB == SPNULL)
196  {
197  #ifdef UG_PARALLEL
198  m_spEigenvector->change_storage_type(PST_CONSISTENT);
199  #endif
200  }
201 
202  // residual = v - v_old
203  VecScaleAdd(*m_spResidual, 1.0, *m_spEigenvectorOld, -1.0, *m_spEigenvector);
204 
205  if(m_spResidual->norm() <= m_dPrecision)
206  {
207  UG_LOG("PowerMethod::calculate_max_eigenvalue() converged after " << m_iteration << " iterations." << std::endl);
208  break;
209  }
210 
211  if(m_iteration == m_maxIterations-1)
212  UG_LOG("PowerMethod::calculate_max_eigenvalue() reached precision of " << m_spResidual->norm() << " after " << m_maxIterations << " iterations." << std::endl);
213  }
214 
215  // lambda = <v,Av>
216  m_spLinOpA->apply(*m_spResidual, *m_spEigenvector);
217  m_dMaxEigenvalue = m_spEigenvector->dotprod(*m_spResidual);
218 
219  return true;
220  }

222  int calculate_min_eigenvalue()
223  {
224  PROFILE_FUNC_GROUP("PowerMethod");
225 
226  UG_COND_THROW(m_spLinOpA == SPNULL, "PowerMethod::calculate_min_eigenvalue(): Linear operator A not set, please specify.");
227  UG_COND_THROW(m_spSolver == SPNULL, "PowerMethod::calculate_min_eigenvalue(): Solver not set, please specify.");
228 
229  m_spResidual = create_approximation_vector();
230 
231  m_spSolver->init(m_spLinOpA);
232 
233  for(m_iteration = 0; m_iteration < m_maxIterations; ++m_iteration)
234  {
235  m_spEigenvectorOld = m_spEigenvector->clone();
236 
237  // residual = B v
238  if(m_spLinOpB != SPNULL)
239  m_spLinOpB->apply(*m_spResidual, *m_spEigenvector);
240  // in case B is not set
241  else
242  {
243  m_spResidual = m_spEigenvector->clone();
244 
245  #ifdef UG_PARALLEL
246  m_spResidual->change_storage_type(PST_ADDITIVE);
247  #endif
248  }
249 
250  // v = A^-1 B v or v = A^-1 v
251  m_spSolver->apply(*m_spEigenvector, *m_spResidual);
252 
253  // v = v / ||v||_B
254  normalize_approximations();
255 
256  // in case B is not set, restore storage type of m_spEigenvector to PST_CONSISTENT
257  // changed by norm calculation in normalize_approximations()
258  if(m_spLinOpB == SPNULL)
259  {
260  #ifdef UG_PARALLEL
261  m_spEigenvector->change_storage_type(PST_CONSISTENT);
262  #endif
263  }
264 
265  // residual = v - v_old
266  VecScaleAdd(*m_spResidual, 1.0, *m_spEigenvectorOld, -1.0, *m_spEigenvector);
267 
268  if(m_spResidual->norm() <= m_dPrecision)
269  {
270  UG_LOG("PowerMethod::calculate_min_eigenvalue() converged after " << m_iteration << " iterations." << std::endl);
271  break;
272  }
273 
274  if(m_iteration == m_maxIterations-1)
275  UG_LOG("PowerMethod::calculate_min_eigenvalue() reached precision of " << m_spResidual->norm() << " after " << m_maxIterations << " iterations." << std::endl);
276  }
277 
278  // lambda = <v,Av>
279  m_spLinOpA->apply(*m_spResidual, *m_spEigenvector);
280  m_dMinEigenvalue = m_spEigenvector->dotprod(*m_spResidual);
281 
282  return true;
283  }
