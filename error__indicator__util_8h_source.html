44 template<typename TElem>
45 number ComputeAvg
46 ( MultiGrid::AttachmentAccessor<TElem, ug::Attachment<number> >& aaError,
47  ConstSmartPtr<DoFDistribution> dd,
48  number& min, number& max, number& sum, size_t& numElem,
49  number& minLocal, number& maxLocal, number& sumLocal, size_t& numElemLocal
50 )
51 {
52  typedef typename DoFDistribution::traits<TElem>::const_iterator const_iterator;
53 
54 // reset maximum of error
55  max = 0.0, min = std::numeric_limits<number>::max();
56  sum = 0.0;
57  number err = 0.0;
58  numElem = 0;
59 
60 // get element iterator
61  const_iterator iter = dd->template begin<TElem>();
62  const_iterator iterEnd = dd->template end<TElem>();
63 
64 // loop all elements to find the maximum of the error
65  for (; iter != iterEnd; ++iter)
66  {
67  // get element
68  TElem* elem = *iter;
69 
70  const number elemEta2 = aaError[elem];
71 
72  // if no error value exists: ignore (might be newly added by refinement);
73  // newly added elements are supposed to have a negative error estimator
74  if (aaError[elem] < 0) UG_THROW("error value invalid!");//continue;
75 
76 
77  const number elemEta = sqrt(aaError[elem]);
78 
79  // search for maximum and minimum
80  if (elemEta > max) max = elemEta;
81  if (elemEta < min) min = elemEta;
82 
83  // sum up total error
84  sum += elemEta;
85  err += elemEta2;
86  ++numElem;
87  }
88 
89  // set local variables
90  maxLocal = max;
91  minLocal = min;
92  sumLocal = sum;
93 #ifdef UG_PARALLEL
94  number errLocal = err;
95 #endif
96  numElemLocal = numElem;
97 
98 #ifdef UG_PARALLEL
99  if (pcl::NumProcs() > 1)
100  {
101  pcl::ProcessCommunicator com;
102  max = com.allreduce(maxLocal, PCL_RO_MAX);
103  min = com.allreduce(minLocal, PCL_RO_MIN);
104  sum = com.allreduce(sumLocal, PCL_RO_SUM);
105  err = com.allreduce(errLocal, PCL_RO_SUM);
106  numElem = com.allreduce(numElemLocal, PCL_RO_SUM);
107  }
108 #endif
109  UG_LOG(" +++++ Error indicator on " << numElem << " elements +++++\n");
110  UG_LOG(" +++ Element errors: maximum=" << max << ", minimum="
111  << min << ", sum=" << sum << ", error=" << err << ".\n");
112 
113  return (sum/numElem);
114 }

206 template<typename TElem>
207 void MarkElements(MultiGrid::AttachmentAccessor<TElem, ug::Attachment<number> >& aaError,
208  IRefiner& refiner,
209  ConstSmartPtr<DoFDistribution> dd,
210  number TOL,
211  number refineFrac, number coarseFrac,
212  int maxLevel)
213 {
214  typedef typename DoFDistribution::traits<TElem>::const_iterator const_iterator;
215 
216 // compute minimal/maximal/ total error and number of elements
217  number min, max, totalErr;
218  size_t numElem;
219  ComputeMinMaxTotal(aaError, dd, min, max, totalErr, numElem);
220 
221 // check if total error is smaller than tolerance. If that is the case we're done
222  if(totalErr < TOL)
223  {
224  UG_LOG(" +++ Total error "<<totalErr<<" smaller than TOL ("<<TOL<<"). done.");
225  return;
226  }
227 
228 // Compute minimum
229  number minErrToRefine = max * refineFrac;
230  UG_LOG(" +++ Refining elements if error greater " << refineFrac << "*" << max <<
231  " = " << minErrToRefine << ".\n");
232  number maxErrToCoarse = min * (1 + coarseFrac);
233  if(maxErrToCoarse < TOL/numElem) maxErrToCoarse = TOL/numElem;
234  UG_LOG(" +++ Coarsening elements if error smaller " << maxErrToCoarse << ".\n");
235 
236 // reset counter
237  int numMarkedRefine = 0, numMarkedCoarse = 0;
238 
239  const_iterator iter = dd->template begin<TElem>();
240  const_iterator iterEnd = dd->template end<TElem>();
241 
242 // loop elements for marking
243  for(; iter != iterEnd; ++iter)
244  {
245  // get element
246  TElem* elem = *iter;
247 
248  // marks for refinement
249  if(aaError[elem] >= minErrToRefine)
250  if(dd->multi_grid()->get_level(elem) <= maxLevel)
251  {
252  refiner.mark(elem, RM_REFINE);
253  numMarkedRefine++;
254  }
255 
256  // marks for coarsening
257  if(aaError[elem] <= maxErrToCoarse)
258  {
259  refiner.mark(elem, RM_COARSEN);
260  numMarkedCoarse++;
261  }
262  }
263 
264 #ifdef UG_PARALLEL
265  if(pcl::NumProcs() > 1)
266  {
267  UG_LOG(" +++ Marked for refinement on Proc "<<pcl::ProcRank()<<": " << numMarkedRefine << " Elements.\n");
268  UG_LOG(" +++ Marked for coarsening on Proc "<<pcl::ProcRank()<<": " << numMarkedCoarse << " Elements.\n");
269  pcl::ProcessCommunicator com;
270  int numMarkedRefineLocal = numMarkedRefine, numMarkedCoarseLocal = numMarkedCoarse;
271  numMarkedRefine = com.allreduce(numMarkedRefineLocal, PCL_RO_SUM);
272  numMarkedCoarse = com.allreduce(numMarkedCoarseLocal, PCL_RO_SUM);
273  }
274 #endif
275 
276  UG_LOG(" +++ Marked for refinement: " << numMarkedRefine << " Elements.\n");
277  UG_LOG(" +++ Marked for coarsening: " << numMarkedCoarse << " Elements.\n");
278 }

380 template<typename TElem>
381 void MarkElementsForCoarsening
382 ( MultiGrid::AttachmentAccessor<TElem,
383  ug::Attachment<number> >& aaError,
384  IRefiner& refiner,
385  ConstSmartPtr<DoFDistribution> dd,
386  number TOL,
387  number coarseFrac,
388  int maxLevel
389 )
390 {
391  typedef typename DoFDistribution::traits<TElem>::const_iterator const_iterator;
392 
393 // compute minimal/maximal/ total error and number of elements
394  number min, max, totalErr;
395  size_t numElem;
396  ComputeMinMaxTotal(aaError, dd, min, max, totalErr, numElem);
397 
398 // compute maximum
399  //number maxErrToCoarse = min * (1+coarseFrac);
400  //if (maxErrToCoarse < TOL/numElem) maxErrToCoarse = TOL/numElem;
401  number maxErrToCoarse = TOL / numElem / 4.0 / 8.0; // = tol_per_elem / (2^2*safetyfactor)
402  UG_LOG(" +++ Coarsening elements if avg child error smaller than "<< maxErrToCoarse << ".\n");
403 
404 // reset counter
405  size_t numMarkedCoarse = 0;
406 
407  const_iterator iter = dd->template begin<TElem>();
408  const_iterator iterEnd = dd->template end<TElem>();
409 
410 // loop elements for marking
411  for (; iter != iterEnd; ++iter)
412  {
413  // get element
414  TElem* elem = *iter;
415 
416  // ignore if already marked for coarsening
417  if (refiner.get_mark(elem) & RM_COARSEN) continue;
418 
419  // get parent
420  TElem* parent = dynamic_cast<TElem*>(dd->multi_grid()->get_parent(elem));
421  if (!parent) continue; // either dynamic casting failed or parent does not exist
422 
423  // sum up error values over all children of parent
424  number sum = 0.0;
425  size_t cnt = 0;
426  size_t nCh = dd->multi_grid()->num_children<TElem>(parent);
427  for (size_t ch = 0; ch < nCh; ch++)
428  {
429  TElem* child = dd->multi_grid()->get_child<TElem>(parent, ch);
430 
431  // if no error value exists: ignore (might be newly added by refinement);
432  // newly added elements are supposed to have a negative error estimator
433  if (aaError[child] < 0) continue;
434 
435  sum += aaError[child];
436  cnt++;
437  }
438 
439  // marks for coarsening
440  if (sum/cnt <= maxErrToCoarse)
441  {
442  for (size_t ch = 0; ch < nCh; ch++)
443  {
444  TElem* child = dd->multi_grid()->get_child<TElem>(parent, ch);
445  if (refiner.get_mark(child) & RM_COARSEN) continue;
446  refiner.mark(child, RM_COARSEN);
447  numMarkedCoarse++;
448  }
449  }
450  }
451 
452 #ifdef UG_PARALLEL
453  if (pcl::NumProcs() > 1)
454  {
455  pcl::ProcessCommunicator com;
456  size_t numMarkedCoarseLocal = numMarkedCoarse;
457  numMarkedCoarse = com.allreduce(numMarkedCoarseLocal, PCL_RO_SUM);
458  }
459 #endif
460 
461  UG_LOG(" +++ Marked for coarsening: " << numMarkedCoarse << " Elements.\n");
462 }
