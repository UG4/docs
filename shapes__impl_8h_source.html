55 template <class vector_t>
56 AABox<vector_t>::AABox(const vector_t* points, size_t numPoints)
57 {
58  if(numPoints < 1){
59  VecSet(min, 0);
60  VecSet(max, 0);
61  return;
62  }
63 
64  min = max = points[0];
65 
66  for(size_t i = 1; i < numPoints; ++i){
67  const vector_t& p = points[i];

69  min[j] = std::min(min[j], p[j]);
70  max[j] = std::max(max[j], p[j]);
71  }
72  }
73 }
