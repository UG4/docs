169 void ISubsetHandler::
170 assign_subset(TIterator iterBegin, TIterator iterEnd, int subsetIndex)
171 {
172  typename TIterator::value_type elem;
173  while(iterBegin != iterEnd)
174  {
175  elem = *iterBegin;
176  ++iterBegin;
177  assign_subset(elem, subsetIndex);
178  }
