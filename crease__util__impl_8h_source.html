50  if(!sel.grid())
51  return;

53  Grid& grid = *sel.grid();
54 
55  number thresholdRad = deg_to_rad(thresholdAngle);
56  AssocElemIter<Vertex, Edge> eiter(cbConsiderEdge);
57 
58  for(TVrtIter viter = vrtsBegin; viter != vrtsEnd; ++viter){
59  Vertex* v = *viter;
60  Vertex* cv[2];
61  int numCons = 0;
62  for(eiter.reinit(grid, v); eiter.valid(); ++eiter){
63  Edge* e = *eiter;
64  if(numCons < 2)
65  cv[numCons] = GetConnectedVertex(e, v);
66  ++numCons;
67  }
68  if(numCons == 2){
69  vector_t dir0, dir1;
70  VecSubtract(dir0, aaPos[v], aaPos[cv[0]]);
71  VecSubtract(dir1, aaPos[cv[1]], aaPos[v]);
72  if(VecAngle(dir0, dir1) >= thresholdRad)
73  sel.select(v);
74  }
75  else if(selectDarts){
76  sel.select(v);
77  }
78  }
