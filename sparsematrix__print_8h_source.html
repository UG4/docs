57 template<typename T>
58 void SparseMatrix<T>::printrow(size_t row) const
59 {
60  UG_LOG("row " << row << ": ");
61  for(const_row_iterator it=begin_row(row); it != end_row(row); ++it)
62  {
63  if(it.value() == 0.0) continue;
64  UG_LOG(" ");
65  UG_LOG("(" << it.index() << " -> " << it.value() << ")");
66  }
