43 template<typename A>
44 inline double BlockNorm2(const DenseMatrix<A> &mat)
45 {
46  double sum=0;
47  for(size_t r=0; r < mat.num_rows(); r++)
48  for(size_t c=0; c < mat.num_cols(); c++)
49  sum += BlockNorm2(mat(r, c));
50 
51  return sum;
52 }

95 }
