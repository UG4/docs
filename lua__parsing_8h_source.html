166 template <>
167 struct LuaParsing<void*>{
168  static bool checkAndGet(std::pair<void*, const ClassNameNode*>& res,
169  lua_State* L, int index, const char* baseClassName){
170  if(!lua_isuserdata(L, index)) return false;
171 
172  UserDataWrapper* udata =
173  reinterpret_cast<UserDataWrapper*>(lua_touserdata(L, index));
174 
175  if(udata->is_const()) return false;
176 
177  // extract the pointer to the object.
178  // udata is either a RawUserData or a SmartUserDataWrapper
179  void* obj = NULL;
180  if(udata->is_raw_ptr())
181  obj = static_cast<RawUserDataWrapper*>(udata)->obj;
182  else if(udata->is_smart_ptr() && IMLPICIT_SMART_PTR_TO_PTR_CONVERSION)
183  obj = static_cast<SmartUserDataWrapper*>(udata)->smartPtr.get();
184  else return false;
185 
186  // get the object and its metatable. Make sure that obj can be cast to
187  // the type that is expected by the paramsTemplate.
188  if(lua_getmetatable(L, index) == 0) return false;
189 
190  lua_pushstring(L, "class_name_node");
191  lua_rawget(L, -2);
192  const ClassNameNode* classNameNode = (const ClassNameNode*) lua_touserdata(L, -1);
193  lua_pop(L, 2);
194 
195  if(!classNameNode) return false;
196  if(classNameNode->empty()) return false;
197  if(!ClassNameTreeContains(*classNameNode, baseClassName)) return false;
198 
199  res.first = obj;
200  res.second = classNameNode;
201 
202  return true;
203  }
204 
205  static void push(lua_State* L, void* data, const char* className){
206  CreateNewUserData(L, data, className, NULL, false);
207  }
208 };
