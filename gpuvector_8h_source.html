111  void assure_on_gpu()
112  {
113  if(on_gpu()) return;
114  if(m_sizeOnGPU != size())
115  {
116  cudaFree(m_devValues);
117  m_devValues = CudaCreateAndCopyToDevice(*this);
118  }
119  else
120  CudaCpyToDevice(m_devValues, *this);
121  m_GPUState = m_GPUState | ON_GPU;
122  }

127  }

129  void assure_on_cpu()
130  {
131  if(on_cpu()) return;
132  // do this before so CudaCpyToHost can access [0] as dest without
133  // calling assure_on_cpu again.
134  m_GPUState = m_GPUState | ON_CPU;
135  CudaCpyToHost(*this, m_devValues);
136  }
