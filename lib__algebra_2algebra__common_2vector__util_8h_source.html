44 template<typename TBlock>
45 bool BlockVectorFiniteAndNotTooBig(TBlock &v, double tooBigValue=1e24)
46 {
47  for(size_t j=0; j< GetSize(v); j++)
48  {
49  double d = BlockRef(v, j);
50  if(d > tooBigValue || d < -tooBigValue || std::isfinite(d) == false)
51  return false;
52  }
53  return true;
54 }
