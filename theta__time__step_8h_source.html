160 template <typename TAlgebra>
161 class ThetaTimeStep
162  : public MultiStepTimeDiscretization<TAlgebra>
163 {
164  public:
166  typedef IDomainDiscretization<TAlgebra> domain_discretization_type;
167 
169  typedef TAlgebra algebra_type;
170 
172  typedef typename algebra_type::matrix_type matrix_type;
173 
175  typedef typename algebra_type::vector_type vector_type;
176 
177  public:
179  ThetaTimeStep(SmartPtr<IDomainDiscretization<TAlgebra> > spDD)
180  : MultiStepTimeDiscretization<TAlgebra>(spDD),
181  m_stage(1), m_scheme("Theta")
182  {
183  set_theta(1.0);
184  this->m_prevSteps = 1;
185  }
186 
188  ThetaTimeStep(SmartPtr<IDomainDiscretization<TAlgebra> > spDD, number theta)
189  : MultiStepTimeDiscretization<TAlgebra>(spDD),
190  m_stage(1), m_scheme("Theta")
191  {
192  set_theta(theta);
193  this->m_prevSteps = 1;
194  }
195 
197  ThetaTimeStep(SmartPtr<IDomainDiscretization<TAlgebra> > spDD, const char* scheme)
198  : MultiStepTimeDiscretization<TAlgebra>(spDD),
199  m_stage(1), m_scheme(scheme)
200  {
201  set_theta(1.0);
202  this->m_prevSteps = 1;
203  }
204 
205  virtual ~ThetaTimeStep() {};
206 
208  void set_scheme(const char* scheme) {m_scheme = scheme;}
209 
211  virtual size_t num_stages() const
212  {
213  if (m_scheme == "Theta") return 1;
214  else if (m_scheme == "Alexander") return 2;
215  else if (m_scheme == "FracStep") return 3;
216  else UG_THROW("Step Scheme not recognized.");
217  }
218 
220  virtual void set_stage(size_t stage) {m_stage = stage;}
221 
223  void set_theta(number theta) {m_theta = theta;}
224 
225  protected:
226  virtual number update_scaling(std::vector<number>& vSM,
227  std::vector<number>& vSA,
228  number dt, number currentTime,
229  ConstSmartPtr<VectorTimeSeries<vector_type> > prevSol)
230  {
231  // resize scaling factors
232  vSM.resize(2);
233  vSM[0] = 1.;
234  vSM[1] = -1.;
235 
236  if(m_scheme == "Theta")
237  {
238  vSA.resize(2);
239  vSA[0] = (m_theta) * dt;
240  vSA[1] = (1.- m_theta) * dt;
241  return currentTime + dt;
242  }
243  else if(m_scheme == "Alexander")
244  {
245  vSA.resize(2);
246  const number gamma = 1 - 1. / sqrt(2.);
247  switch(m_stage)
248  {
249  case 1:
250  vSA[0] = gamma * dt;
251  vSA[1] = 0;
252  return currentTime + gamma * dt;
253  case 2:
254  vSA[0] = gamma * dt;
255  vSA[1] = (1.- 2*gamma) * dt;
256  return currentTime + (1 - gamma) * dt;
257  default:
258  UG_THROW("Alexander scheme has only 2 stages")
259  }
260  }
261  else if(m_scheme == "FracStep")
262  {
263  vSA.resize(2);
264  switch(m_stage)
265  {
266  case 1:
267  vSA[0] = (2.-sqrt(2.)) * (1-1./sqrt(2.)) * dt;
268  vSA[1] = (1.- (2.-sqrt(2.)))* (1-1./sqrt(2.)) * dt;
269  return currentTime + (1-1./sqrt(2.)) * dt;
270  case 2:
271  vSA[0] = (sqrt(2.)-1) * (sqrt(2.)-1) * dt;
272  vSA[1] = (1.- (sqrt(2.)-1)) * (sqrt(2.)-1) * dt;
273  return currentTime + (sqrt(2.)-1) * dt;
274  case 3:
275  vSA[0] = (2.-sqrt(2.)) * (1-1./sqrt(2.)) * dt;
276  vSA[1] = (1.- (2.-sqrt(2.)))* (1-1./sqrt(2.)) * dt;
277  return currentTime + (1-1./sqrt(2.)) * dt;
278  default:
279  UG_THROW("FracStep scheme has only 3 stages")
280  }
281  }
282  else
283  UG_THROW("Unknown Multi-Stage Theta Scheme: "<< m_scheme<<".");
284 
285  }
286 
287  number m_theta;
288 
289  size_t m_stage;
290 
291  std::string m_scheme;
292 };

295 template <typename TAlgebra>
296 class BDF
297  : public MultiStepTimeDiscretization<TAlgebra>
298 {
299  public:
301  typedef IDomainDiscretization<TAlgebra> domain_discretization_type;
302 
304  typedef TAlgebra algebra_type;
305 
307  typedef typename algebra_type::matrix_type matrix_type;
308 
310  typedef typename algebra_type::vector_type vector_type;
311 
312  public:
314  BDF(SmartPtr<IDomainDiscretization<TAlgebra> > spDD)
315  : MultiStepTimeDiscretization<TAlgebra>(spDD)
316  {
317  set_order(2);
318  }
319 
321  BDF(SmartPtr<IDomainDiscretization<TAlgebra> > spDD, size_t order)
322  : MultiStepTimeDiscretization<TAlgebra>(spDD)
323  {
324  set_order(order);
325  }
326 
327  virtual ~BDF() {};
328 
330  void set_order(size_t order) {m_order = order; this->m_prevSteps = order;}
331 
333  virtual size_t num_stages() const {return 1;}
334 
336  virtual void set_stage(size_t stage)
337  {
338  if(stage!=1) UG_THROW("BDF has only one stage.");
339  }
340 
341  protected:
342  virtual number update_scaling(std::vector<number>& vSM,
343  std::vector<number>& vSA,
344  number dt, number currentTime,
345  ConstSmartPtr<VectorTimeSeries<vector_type> > prevSol)
346  {
347  // resize scaling factors
348  vSM.resize(m_order+1);
349  vSA.resize(m_order+1);
350 
351  // future time
352  const number futureTime = currentTime + dt;
353 
354  // get time points
355  if(prevSol->size() < m_order)
356  UG_THROW("BDF("<<m_order<<") needs at least "<< m_order <<
357  " previous solutions, but only "<<prevSol->size()<<"passed.");
358 
359  std::vector<number> vTimePoint(m_order+1);
360  vTimePoint[0] = futureTime;
361  for(size_t i = 1; i <= m_order; ++i)
362  vTimePoint[i] = prevSol->time(i-1);
363 
364  // evaluate derivative of Lagrange Polynoms at future time
365  for(size_t i = 0; i <= m_order; ++i)
366  {
367  vSM[i] = 0;
368 
369  for(size_t j = 0; j < vTimePoint.size(); ++j)
370  {
371  if(j == i) continue;
372 
373  number prod = 1;
374  for(size_t k = 0; k < vTimePoint.size(); ++k)
375  {
376  if(k == i) continue;
377  if(k == j) continue;
378 
379  prod *= (vTimePoint[0]-vTimePoint[k])/
380  (vTimePoint[i]-vTimePoint[k]);
381  }
382  prod *= 1./(vTimePoint[i]-vTimePoint[j]);
383 
384  vSM[i] += prod;
385  }
386  }
387 
388  // only first value of vSA != 0
389  const number scale = 1.0 / vSM[0];
390  vSA[0] = scale;
391  for(size_t i = 1; i <= m_order; ++i) vSA[i] = 0;
392 
393  // scale first s_m to 1.0
394  for(int i = m_order; i >= 0; --i) vSM[i] *= scale;
395 
396  return futureTime;
397  }
398 
399  size_t m_order;
400 };
