97  StringTableStream &operator << (const char *c)
98  {
99  size_t l = strlen(c);
100 // size_t begin=0;
101  std::vector<char> vc;
102  for(size_t i=0; i<l; i++)
103  {
104  if(c[i] == '\n')
105  {
106  vc.push_back(0);
107  s(m_curRow, m_curCol) = &vc[0];
108  vc.clear();
109  m_curRow++;
110  m_curCol=0;
111  if(i == l-1) return *this;
112  }
113  else
114  vc.push_back(c[i]);
115  }
116  vc.push_back(0);
117  s(m_curRow, m_curCol++) = &vc[0];
118  return *this;
119  }
