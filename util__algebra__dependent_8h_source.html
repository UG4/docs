129  template <typename TAlgebra>
130  size_t id()
131  {
132  // maybe this can be done faster
133  // however, unless number of algebra types grows significantly,
134  // this should be fine
135  for (size_t i = 0; i < NUM_ALGEBRA_TYPES; ++i)
136  {
137  AlgebraID<TAlgebra>* p_aid = dynamic_cast<AlgebraID<TAlgebra>*>(m_aid[i]);
138  if (p_aid) return i;
139  }
140  UG_THROW("Cannot provide Algebra type index. Algebra type unknown.");
141  return 0;
142  }
143 
144  private:
145  // helper struct for filling storage of indices via boost::mpl::list
146  template <typename List = CompileAlgebraList>
147  struct RegisterAlgebraIndices

152  (typename boost::mpl::if_c<isEmpty, RegEnd, RegNext>::type (atidp));
