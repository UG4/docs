154  virtual bool preprocess(SmartPtr<MatrixOperator<matrix_type, vector_type> > pOp)
155  {
156 
157  PROFILE_BEGIN_GROUP(Jacobi_preprocess, "algebra Jacobi");
158 
159  matrix_type &mat = *pOp;
160  // create help vector to apply diagonal
161  size_t size = mat.num_rows();
162  if(size != mat.num_cols())
163  {
164  UG_LOG("Square Matrix needed for Jacobi Iteration.\n");
165  return false;
166  }
167 
168  // resize
169  m_diagInv.resize(size);
170 #ifdef UG_PARALLEL
171  // temporary vector for the diagonal
172  ParallelVector<Vector< typename matrix_type::value_type > > diag;
173  diag.resize(size);
174 
175  // copy the layouts+communicator into the vector
176  diag.set_layouts(mat.layouts());
177 
178  // copy diagonal
179  for(size_t i = 0; i < diag.size(); ++i){
180  diag[i] = mat(i, i);
181  }
182 
183  // make diagonal consistent
184  diag.set_storage_type(PST_ADDITIVE);
185  diag.change_storage_type(PST_CONSISTENT);
186 
187 
188  if(diag.size() > 0)
189  if(CheckVectorInvertible(diag) == false)
190  return false;
191 // UG_ASSERT(CheckVectorInvertible(diag), "Jacobi: A has noninvertible diagonal");
192 
193 #endif
194 
195 // get damping in constant case to damp at once
196  number damp = 1.0;
197  if(damping()->constant_damping())
198  damp = damping()->damping();
199 
200  typename matrix_type::value_type m;
201  // invert diagonal and multiply by damping
202  for(size_t i = 0; i < mat.num_rows(); ++i)
203  {
204 #ifdef UG_PARALLEL
205  typename matrix_type::value_type &d = diag[i];
206 #else
207  typename matrix_type::value_type &d = mat(i,i);
208 #endif
209  if(m_bBlock)
210  GetDiag(m, d);
211  else
212  m = d;
213  m *= 1./damp;
214  GetInverse(m_diagInv[i], m);
215  }
216 
217  // done
218  return true;
219  }

221  virtual bool step(SmartPtr<MatrixOperator<matrix_type, vector_type> > pOp, vector_type& c, const vector_type& d)
222  {
223  PROFILE_BEGIN_GROUP(Jacobi_step, "algebra Jacobi");
224 
225  // multiply defect with diagonal, c = damp * D^{-1} * d
226  // note, that the damping is already included in the inverse diagonal
227  for(size_t i = 0; i < m_diagInv.size(); ++i)
228  {
229  // c[i] = m_diagInv[i] * d[i];
230  MatMult(c[i], 1.0, m_diagInv[i], d[i]);
231  }
232 
233 #ifdef UG_PARALLEL
234 
235  // the computed correction is additive
236  c.set_storage_type(PST_ADDITIVE);
237 
238  // we make it consistent
239  if(!c.change_storage_type(PST_CONSISTENT))
240  {
241  UG_LOG("ERROR in 'JacobiPreconditioner::apply': "
242  "Cannot change parallel status of correction to consistent.\n");
243  return false;
244  }
245 #endif
246  // done
247  return true;
248  }

259  if(!this->m_bInit)
260  {
261  UG_LOG("ERROR in '"<<name()<<"::apply': Iterator not initialized.\n");
262  return false;
263  }
264 
265  // Check parallel status
266  #ifdef UG_PARALLEL
267  if(!d.has_storage_type(PST_ADDITIVE))
268  UG_THROW(name() << "::apply: Wrong parallel "
269  "storage format. Defect must be additive.");
270  #endif
271 
272  // Check sizes
273  THROW_IF_NOT_EQUAL_4(c.size(), d.size(), approx_operator()->num_rows(), approx_operator()->num_cols());
274 
275  // apply iterator: c = B*d
276  if(!step(approx_operator(), c, d))
277  {
278  UG_LOG("ERROR in '"<<name()<<"::apply': Step Routine failed.\n");
279  return false;
280  }
281 
282  // apply scaling
283  if(!damping()->constant_damping()){
284  const number kappa = damping()->damping(c, d, approx_operator());
285  if(kappa != 1.0){
286  c *= kappa;
287  }
288  }
289 
290  // Correction is always consistent
291  #ifdef UG_PARALLEL
292  if(!c.change_storage_type(PST_CONSISTENT))
293  UG_THROW(name() << "::apply': Cannot change "
294  "parallel storage type of correction to consistent.");
295  #endif
296 
297  // we're done
298  return true;
299  }
