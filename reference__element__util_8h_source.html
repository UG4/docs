40 inline int ReferenceElementDimension(ReferenceObjectID roid)
41 {
42  switch(roid)
43  {
44  case ROID_VERTEX: return 0;
45  case ROID_EDGE: return 1;
46  case ROID_TRIANGLE: return 2;
47  case ROID_QUADRILATERAL: return 2;
48  case ROID_TETRAHEDRON: return 3;
49  case ROID_OCTAHEDRON: return 3;
